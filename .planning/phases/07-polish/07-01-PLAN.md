Modify the Home RSC function in page.tsx to perform drift recalculation before rendering:

1. Import `and`, `lt`, `isNotNull`, `sql` from `drizzle-orm` (in addition to existing `eq`).

2. Before the existing `db.select()`, add drift increment logic:
   ```typescript
   const now = new Date();

   // Increment driftCount for active tasks whose horizon window has passed
   // Also advance targetDateLatest to prevent double-counting on next refresh
   const drifted = await db.update(tasks)
     .set({
       driftCount: sql`${tasks.driftCount} + 1`,
       // Advance targetDateLatest by the window duration to prevent double-counting.
       // Uses the difference between earliest and latest as the window width,
       // falling back to 7 days if both dates are the same.
       targetDateLatest: sql`NOW() + GREATEST(
         (${tasks.targetDateLatest} - ${tasks.targetDateEarliest}),
         INTERVAL '7 days'
       )`,
       targetDateEarliest: sql`NOW()`,
     })
     .where(and(
       eq(tasks.status, 'active'),
       isNotNull(tasks.targetDateLatest),
       lt(tasks.targetDateLatest, now),
     ))
     .returning();
   ```

3. After the drift update, also auto-flag `needsRefinement` for tasks that just hit 3+ drifts (if not already flagged):
   ```typescript
   // Auto-flag needsRefinement at 3+ drifts (generates prompt lazily when user opens detail)
   if (drifted.length > 0) {
     const newlyHighDrift = drifted.filter(t => t.driftCount >= 3 && !t.needsRefinement);
     if (newlyHighDrift.length > 0) {
       await db.update(tasks)
         .set({ needsRefinement: true })
         .where(
           sql`${tasks.id} IN (${sql.join(newlyHighDrift.map(t => sql`${t.id}`), sql`, `)})`
         );
     }
   }
   ```

4. Keep the existing `db.select()` for allTasks AFTER the drift update so the returned data includes updated driftCounts.

5. Pass `driftSummary` to SceneLoader:
   ```typescript
   return (
     <SceneLoader
       initialTasks={allTasks}
       driftSummary={drifted.length > 0 ? { count: drifted.length } : null}
     />
   );
   ```

IMPORTANT: The `targetDateLatest` advance is the double-count prevention mechanism. By moving it forward, the `lt(tasks.targetDateLatest, now)` check won't trigger again until the new window passes. Someday tasks have null targetDateLatest, so they're excluded by the `isNotNull` check.

'use client';
import { useState, useEffect } from 'react';

interface DriftNotificationProps {
  count: number;
}

export function DriftNotification({ count }: DriftNotificationProps) {
  const [visible, setVisible] = useState(true);
  const [fading, setFading] = useState(false);

  useEffect(() => {
    const fadeTimer = setTimeout(() => setFading(true), 4000);
    const removeTimer = setTimeout(() => setVisible(false), 5000);
    return () => { clearTimeout(fadeTimer); clearTimeout(removeTimer); };
  }, []);

  if (!visible) return null;

  return (
    <>
      <style>{`
        @keyframes driftToastIn { from { opacity: 0; transform: translate(-50%, -20px); } to { opacity: 1; transform: translate(-50%, 0); } }
        @keyframes driftToastOut { from { opacity: 1; } to { opacity: 0; } }
      `}</style>
      <div style={{
        position: 'fixed',
        top: 24,
        left: '50%',
        transform: 'translateX(-50%)',
        zIndex: 105,
        padding: '10px 20px',
        background: 'rgba(245, 158, 11, 0.12)',
        border: '1px solid rgba(245, 158, 11, 0.3)',
        borderRadius: 24,
        backdropFilter: 'blur(12px)',
        WebkitBackdropFilter: 'blur(12px)',
        fontFamily: 'monospace',
        fontSize: 13,
        color: '#f59e0b',
        animation: fading ? 'driftToastOut 1s ease forwards' : 'driftToastIn 0.3s ease forwards',
        whiteSpace: 'nowrap',
      }}>
        {count} task{count !== 1 ? 's' : ''} drifted past {count !== 1 ? 'their' : 'its'} horizon
      </div>
    </>
  );
}
