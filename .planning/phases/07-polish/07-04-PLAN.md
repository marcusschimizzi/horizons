**ListView.tsx (NEW):**

Create a full list view component with the following structure:

2. Constants for horizon display:
```typescript
const HORIZON_ORDER: Horizon[] = ['immediate', 'this-week', 'this-month', 'this-quarter', 'this-year', 'someday'];
const HORIZON_LABELS: Record<string, string> = { ... };  // Same as TaskDetail
const HORIZON_COLORS: Record<string, string> = { ... };  // Same as TaskDetail
```

3. Component structure:
```tsx
export function ListView() {
  const tasks = useTasksWithHorizon();
  const store = useContext(TaskStoreContext);
  const listFilters = useTaskStore((s) => s.listFilters);
  const [collapsedSections, setCollapsedSections] = useState<Set<string>>(new Set());
  const [rescheduleOpenId, setRescheduleOpenId] = useState<string | null>(null);

  // Filter tasks
  const filteredTasks = useMemo(() => {
    return tasks.filter(task => {
      // Tag filter
      if (listFilters.tags.size > 0) {
        const taskTags = task.tags || [];
        if (!taskTags.some(t => listFilters.tags.has(t))) return false;
      }
      // Needs refinement filter
      if (listFilters.needsRefinement === true && !task.needsRefinement) return false;
      if (listFilters.needsRefinement === false && task.needsRefinement) return false;
      // Horizon filter
      if (listFilters.horizons.size > 0 && !listFilters.horizons.has(task.horizon)) return false;
      return true;
    });
  }, [tasks, listFilters]);

  // Group by horizon
  const grouped = useMemo(() => {
    const map = new Map<string, Task[]>();
    for (const h of HORIZON_ORDER) map.set(h, []);
    for (const task of filteredTasks) {
      map.get(task.horizon)?.push(task);
    }
    return map;
  }, [filteredTasks]);

  // Toggle section collapse
  const toggleSection = (horizon: string) => {
    setCollapsedSections(prev => {
      const next = new Set(prev);
      if (next.has(horizon)) next.delete(horizon); else next.add(horizon);
      return next;
    });
  };

  // Quick actions
  const handleQuickComplete = (taskId: string) => { /* same logic as TaskDetail handleComplete but simplified: startCompletion, schedule PATCH, finishCompletion */ };
  const handleQuickDrop = (taskId: string) => { /* same as TaskDetail handleDrop */ };
  const handleQuickReschedule = (taskId: string, horizon: Horizon) => { /* same as TaskDetail handleReschedule */ };

  return (
    <div style={containerStyle}>
      {/* Filter bar */}
      <div style={filterBarStyle}>
        {/* Tag filters */}
        {['work', 'personal', 'health', 'finance', 'home', 'social'].map(tag => (
          <button key={tag} style={filterPillStyle(listFilters.tags.has(tag))} onClick={() => toggleTagFilter(tag)}>
            {tag}
          </button>
        ))}
        {/* Needs refinement toggle */}
        <button style={filterPillStyle(listFilters.needsRefinement === true)} onClick={toggleRefinementFilter}>
          Needs refinement
        </button>
      </div>

      {/* Horizon groups */}
      {HORIZON_ORDER.map(horizon => {
        const sectionTasks = grouped.get(horizon) || [];
        if (sectionTasks.length === 0) return null;
        const isCollapsed = collapsedSections.has(horizon);
        const color = HORIZON_COLORS[horizon];

        return (
          <div key={horizon} style={sectionStyle}>
            <div style={sectionHeaderStyle(color)} onClick={() => toggleSection(horizon)}>
              <span>{isCollapsed ? '\u25B6' : '\u25BC'}</span>
              <span>{HORIZON_LABELS[horizon]}</span>
              <span style={countStyle}>{sectionTasks.length}</span>
            </div>
            {!isCollapsed && sectionTasks.map(task => (
              <div key={task.id} style={taskRowStyle}>
                <div style={taskTitleStyle} onClick={() => store?.getState().selectTask(task.id)}>
                  {task.title}
                  {task.needsRefinement && <span style={refinementDotStyle} />}
                </div>
                {task.driftCount > 0 && <span style={driftBadgeSmallStyle}>{task.driftCount}</span>}
                {task.tags?.map(tag => <span key={tag} style={tagPillStyle}>{tag}</span>)}
                <div style={quickActionsStyle}>
                  <button style={quickBtnStyle('#22c55e')} onClick={() => handleQuickComplete(task.id)}>Done</button>
                  <button style={quickBtnStyle('#94a3b8')} onClick={() => handleQuickDrop(task.id)}>Drop</button>
                  <button style={quickBtnStyle('#3b82f6')} onClick={() => setRescheduleOpenId(rescheduleOpenId === task.id ? null : task.id)}>Move</button>
                </div>
                {/* Inline reschedule dropdown */}
                {rescheduleOpenId === task.id && (
                  <div style={rescheduleDropdownStyle}>
                    {HORIZON_ORDER.map(h => (
                      <button key={h} style={rescheduleOptionStyle(h === task.horizon)} onClick={() => { handleQuickReschedule(task.id, h); setRescheduleOpenId(null); }}>
                        {HORIZON_LABELS[h]}
                      </button>
                    ))}
                  </div>
                )}
              </div>
            ))}
          </div>
        );
      })}

      {filteredTasks.length === 0 && (
        <div style={emptyStyle}>No tasks match the current filters</div>
      )}
    </div>
  );
}
```

**Styling guidelines:**
- Container: `position: 'relative'`, z-index 1, full viewport height, overflow-y auto, dark background matching app (#0a0a0f), padding 20px.
- Filter bar: sticky top, flex row, gap 8, padding 16px 20px, border-bottom.
- Section headers: clickable, flex row with arrow + label + count, colored left border matching horizon color.
- Task rows: flex row, padding 12px 20px, border-bottom, hover effect.
- Quick action buttons: small, borderless, colored text, flex-shrink 0.
- All styles are inline (consistent with the codebase pattern).
- Monospace font throughout to match app aesthetic.

**Quick action implementations:**
- `handleQuickComplete`: Call `store.getState().startCompletion(id)`, then PATCH status to 'completed', then `finishCompletion(id)` after a short delay. No undo in list view (simpler flow).
- `handleQuickDrop`: Call `store.getState().startDrop(id)`, PATCH status to 'dropped', `finishDrop(id)`.
- `handleQuickReschedule`: Use `horizonToDateRange(horizon)` to get new dates, call `store.getState().updateTask(id, { targetDateEarliest, targetDateLatest })`, PATCH to server.
